function buzzIt(valueOne, valueTwo) {
    for (i = 1; i <= 100; i++) {
        let fizz = false;
        let buzz = false;

        if (i % valueOne == 0) {
            fizz = true;
        }
        if (i % valueTwo == 0) {
            buzz = true;
        }

        //the tagIt function tags the corresponding element in the UI with Fizz, Buzz, or FizzBuzz.
        tagIt(i, [fizz, buzz])

    }
}
